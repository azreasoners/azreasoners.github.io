94  // scanner conditions.
95  if (macro_type == M_PAREN_R) lemon_macroAttemptReduce(_parser, this);

97  }
98  } while (!_tokens.size() && _stat == Status::OK);
99

102  token = new Token(T_EOF, new ReferencedString("<EOF>"), _scanner->loc(), _scanner->loc());
103  } else if (_stat == Status::END_INPUT || _stat == Status::OK) {
104  token = _tokens.front();
105  _tokens.pop_front();
106  } else {
107  token = new Token(T_ERR_SYNTAX, new ReferencedString("<ERR_SYNTAX>"), _scanner->loc(), _scanner->loc());
108  }
109  }
110 #ifndef NDEBUG

112 #endif

183 #endif

185  Location l = stmt->beginLoc();
186 
187  BOOST_FOREACH(ReferencedString const* name, *stmt) {
188  fs::path p = fs::path(*name);
189  try {
190  // Try to resolve the name
191  fs::path fullpath;
192 
193  // try to resolve the path of the file
194 
195  if (p.is_absolute() || !l.file()) {
196  fullpath = p;
197  } else if (!l.file()) {
198  fullpath = fs::current_path() / p;
199  } else {
200  // Try a path relative to the current file we're in first
201  // Then try resolving with our default path.
202  fullpath = l.file()->parent_path() / p;
203  if (!fs::exists(fullpath) || fs::is_directory(fullpath)) {
204  fullpath = fs::current_path() / p;

206  }

212 
213  _stat = Status::ERR_SYNTAX;
214  break;
215  }
216 
217  if (fs::is_directory(fullpath)) {
218  // We were unable to resolve the file..
