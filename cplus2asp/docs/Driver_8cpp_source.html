135 /*
136  std::cout << "Entering user-interactive mode. " << std::endl
137  << "NOTICE: The reactive bridge is in alpha and is not yet complete." << std::endl
138  << "It's currently required that the user provide input directly in the input language of oClingo." << std::endl
139  << "Valid oClingo input consists of four components:" << std::endl
140  << " - A step declaration '#step t.' where t is a positive integer. " << std::endl
141  << " This specifies the current step." << std::endl
142  << " - A number of facts specifying the abnormality constants that have occurred." << std::endl
143  << " This should be of one of the forms:" << std::endl
144  << " 'ab_h(<static-abnormality>, t).' OR" << std::endl
145  << " 'ab_occ(<dynamic-abnormality>,t-1).'" << std::endl
146  << " where t is the step provided in the step declaration." << std::endl
147  << " - A line containing only '#endstep.' specifying the end of input. " << std::endl
148  << "When you're finished, you should input 'exit' on a line by itself." << std::endl;
149 */
150 
151 
152  std::cout << "% Attempting to connect to oClingo..." << std::endl;
153 
154  mClient = new NetworkClient(
155  mHost,
156  mPort,
157  boost::bind(&Driver::processOclingoPacket, this, _1, _2, _3));
158 
159  while (!mClient->open()) {
160  boost::this_thread::sleep( boost::posix_time::milliseconds(200));
161 
162  }
163 
164  std::cout << "% Connected succesfully." << std::endl
165  << "% Please provide an observation stream to send to oclingo." << std::endl
166  << "% An observation stream component is of the form:" << std::endl
167  << "% \t#step." << std::endl
168  << "% \t<rules>"<< std::endl
169  << "% \t#endstep."<<std::endl
170  << "% where <rules> is a set of observational laws " << std::endl
171  << "% \tobserved F at t." << std::endl
172  << "% and temporal constraint laws " << std::endl
173  << "% \tconstraint F at t." << std::endl
174  << "% containing no variables." << std::endl
175  << "% At any time you can input the line \"#done.\" to finish and exit." << std::endl << std::endl;
176 
177  std::string line;
178  std::string trimmed;
179 
180 // Providied by initial oClingo prompt.
181 // std::cout << "?- ";
182 
183  while (!mClosing && mClient->open() && std::getline(std::cin, line))
184  {
185 
186  trimmed = line;
187  boost::trim(trimmed);
188  if (trimmed == "#step.") {
189  mTransIn.str("");
190  bool endstep = false, empty = true;
191 
192  // got a new step... collect the data until they say to stop...
193  while (!mClosing && !endstep && mClient->open() && std::getline(std::cin, line)) {
194  trimmed = line;
195  boost::trim(trimmed);
196  if (trimmed == "#endstep.") {
197  endstep = true;
198  } else if (trimmed == "#done.") {
199  endstep = true;
200  mClosing = true;
201  } else {
202  empty = false;
203  // add the line to our input to parse
204  mTransIn << line << std::endl;
205  }
206  }
207 
208  if (mClosing) continue;
209 
210 
211  // parse the input
212  if (!empty) {
213  std::string transin = mTransIn.str();
214  mTransOut.str("");
215  mParser->reset();
216  mParser->push_back(transin.c_str());
217 
218 
219  pa::BCParser::ParseType result;
220  bool ret = true;
221  int step = 0;
222 
223  do {
224  result = mParser->parse();
225 
226  if (result.first == pa::BCParser::Status::IO_ERR ||
227  result.first == pa::BCParser::Status::SYNTAX_ERR) ret = false;
228 
229  if (result.second) {
230  u::ref_ptr<st::Statement> stmt = result.second;
231 
232  switch (stmt->type()) {
233  case st::Statement::Type::LAW_OBSERVED:
234  {
235  u::ref_ptr<const st::ObservedLaw> l = (st::ObservedLaw const*)stmt.get();
236  int step2 = l->at()->val();
237  if (l->head()->cmask() & sy::ConstantSymbol::Type::M_ACTIONS) step2++;
238  step = (step2 > step ? step2 : step);
239  }
240  break;
241  case st::Statement::Type::LAW_TEMPORAL_CONSTRAINT:
242  {
243  u::ref_ptr<const st::TemporalConstraintLaw> l = (st::TemporalConstraintLaw const*)stmt.get();
244  int step2 = l->at()->val();
245  if (l->head()->cmask() & sy::ConstantSymbol::Type::M_ACTIONS) step2++;
246  step = (step2 > step ? step2 : step);
247  }
248  break;
249  default:
250  break;
251  }
252 
253  ret = mTranslator->translate(stmt) && ret;
254  }
255 
256  } while (result.first != pa::BCParser::Status::END_INPUT);
257 
258 
259 
260 
261  // process it through F2LP
262  if (ret) {
263 
264  bp::context ctx;
265  ctx.environment = bp::self::get_environment();
266  ctx.stdin_behavior = bp::capture_stream();
267  ctx.stdout_behavior = bp::capture_stream();
268  ctx.stderr_behavior = bp::inherit_stream();
269 
270  std::list<std::string> arglist;
271  arglist.push_back("-c");
272  arglist.push_back("-i");
273  bp::child c = bp::launch(bp::find_executable_in_path("f2lp"), arglist, ctx);
274 
275  // send our translation data as input
276 /*
277 E
278  copy(std::istreambuf_iterator<char>(mTransOut), std::istreambuf_iterator<char>(),
279  std::ostreambuf_iterator<char>(c.get_stdin()));
280 */
281 
282  boost::thread workerthread(boost::bind(&copy_and_close, &mTransOut, &c.get_stdin()));
283 
284 
285  mF2lpOut.str("");
286 
287  // receive data back as input
288  copy(std::istreambuf_iterator<char>(c.get_stdout()), std::istreambuf_iterator<char>(),
289  std::ostreambuf_iterator<char>(mF2lpOut));
290 
291 
292  // wait for F2LP to finish
293  workerthread.join();
294 
295  bp::status s = c.wait();
296  if (s.exited() && s.exit_status()) ret = false;
297 
298  }
299 
300  // sanitize F2LP output for use with oClingo
301  std::stringstream tmp;
302 
303  std::string line;
304  while (mF2lpOut.good()) {
305  std::getline(mF2lpOut, line);
306  boost::trim(line);
307  if (line[0] == '%' || line[0] == '#') continue;
308  if (line.find(":- false.") == 0) continue;
309  if (line.find("true.") == 0) continue;
310 // if (line.find("false") == 0) line = line.substr(5);
311  tmp << line << "\n";
312  }
313 
314 
315  if (!ret) {
316  std::cerr << "% ERROR: An error occurred parsing the online module. The module will be ignored." << std::endl;
317  } else {
318  send(tmp.str(), step, mClosing);
319  }
320  } else {
321  send("", 0, mClosing);
322  }
323 
324 
325 
326 
327 
328  } else if (trimmed == "#done.") {
329  mClosing = true;
330  } else {
331  std::cerr << "% ERROR: Ignoring invalid input. Expected one of \"#step.\" or \"#done.\"." << std::endl;
332  }
333 
334  }
335 
336  close(0);
337  }
338  catch (std::exception& e)
339  {
340  std::cerr << "Error: " << e.what() << std::endl;
341  return 1;
342  }
343  return 0;
344 }

347 void Driver::processOclingoPacket(boost::system::error_code const& error, char const* msg, size_t length) {
348  std::string strmsg(msg, length), tmp;
349  size_t pos;
350 
351  if (!error) {
352 
353  if (((pos = strmsg.find("Step:")) != std::string::npos)
354  || ((pos = strmsg.find("Input:")) != std::string::npos)) {
355  if (( pos = strmsg.find('\n', pos)) != std::string::npos && pos < strmsg.size())
356  strmsg = strmsg.substr(pos+1);
357  else
358  strmsg = "";
359  mOclingoOut << std::endl;
360  displayTransition(mOclingoOut, std::cout);
361  mOclingoOut.str("");
362  }
363 
364 
365  if ((pos = strmsg.find("End of Step.")) == std::string::npos)
366  mOclingoOut << strmsg;
367  else {
368  tmp = strmsg.substr(0,pos);
369  mOclingoOut << tmp << std::endl;
370  displayTransition(mOclingoOut, std::cout);
371 
372  // Start collecting the next answer set.
373  mOclingoOut.str("");
374  pos += strlen("End of Step.");
375  if (pos <= strmsg.size())
376  mOclingoOut << strmsg.substr(pos);
377  }
378 
379 
380  } else if (!mClosing) {
381  std::cerr << "% The connection to oClingo has been closed remotely.\n";
382  close(1);
383  }
384 
385  if (mClosing) {
386  std::cout << "% Finished." << std::endl;
387  close(0);
388  }
389 // int stat = system(tmpCmd.c_str());
390 
391 }
