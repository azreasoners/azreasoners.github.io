1118 bool Translator::translateCausalLaw(el::Formula const* head, el::Formula const* ifbody, el::Formula const* ifcons, el::Formula const* after, el::AtomicFormula const* unless, el::Formula const* where, StatementList* preStmts, ClauseList* extraClauses, std::stringstream& tmpout) {
1119 
1120 
1121 
1123  struct LawType {
1124  enum type {
1125  STATIC,
1126  ACTION_DYNAMIC,
1127  FLUENT_DYNAMIC,
1128  RIGID
1129  };
1130  };
1131 
1132  LawType::type lt = LawType::STATIC;
1133 
1134  // Ensure that forbidden constants aren't in the head of this law
1135  // These are: incremental and external constants
1136  if (head->cmask() & sy::ConstantSymbol::Type::M_ADDITIVE) {
1137  config()->error("Malformed law: Additive constants can only occur in the heads of \"increments\" laws.", &head->beginLoc());
1138  return false;
1139  }
1140 
1141  if (head->cmask() & sy::ConstantSymbol::Type::M_EXTERNAL) {
1142  config()->error("Malformed law: External constants cannot occur in the heads of any law within the main program.", &head->beginLoc());
1143  return false;
1144  }
1145 
1146 
1147 
1148  // figure out what type of law this is...
1149 
1150  // rigid
1151  if (head->cmask() & sy::ConstantSymbol::Type::RIGID) {
1152  lt = LawType::RIGID;
1153  }
1154 
1155  // rigid constraints
1156  else if (
1157  !(head->cmask() & ~sy::ConstantSymbol::Type::RIGID)
1158  && (!ifbody || !(ifbody->cmask() & ~sy::ConstantSymbol::Type::RIGID))
1159  && (!ifcons || !(ifcons->cmask() & ~sy::ConstantSymbol::Type::RIGID))
1160  && (!where || !(where->cmask() & ~sy::ConstantSymbol::Type::RIGID))
1161  && !after && !unless) {
1162  lt = LawType::RIGID;
1163  }
1164 
1165  // fluent dynamic
1166  else if (!(head->cmask() & sy::ConstantSymbol::Type::M_ACTIONS) && (after || unless)) {
1167  lt = LawType::FLUENT_DYNAMIC;
1168  }
1169 
1170  // action dynamic
1171  else if (head->cmask() & sy::ConstantSymbol::Type::M_ACTIONS) {
1172  lt = LawType::ACTION_DYNAMIC;
1173  }
1174 
1175 
1176 
1177  // Ensure integrity of the law according to law type
1178  switch (lt) {
1179  case LawType::RIGID:
1180 
1181  // Cannot contain an 'after' or 'unless' clause
1182  if (after || unless) {
1183  config()->error("Malformed law: Rigid constants cannot occur in the head of dynamic laws.", &head->beginLoc());
1184  return false;
1185  }
1186 
1187 
1188  // Cannot contain non-rigid constants in any other clause
1189  if (
1190  (ifbody && (ifbody->cmask() & ~sy::ConstantSymbol::Type::RIGID))
1191  || (ifcons && (ifcons->cmask() & ~sy::ConstantSymbol::Type::RIGID))
1192  || (where && (where->cmask() & ~sy::ConstantSymbol::Type::RIGID))
1193  ) {
1194  config()->error("Malformed law: Rigid constants cannot occur in the head of a law containing non-rigid constants.", &head->beginLoc());
1195  return false;
1196  }
1197 
1198  break;
1199  case LawType::ACTION_DYNAMIC:
1200 
1201  // Cannot contain an 'after' or 'unless' clause
1202  if (after || unless) {
1203  config()->error("Malformed law: Action dynamic laws cannot contain \"after\" or \"unless\" clauses.", &head->beginLoc());
1204  return false;
1205  }
1206 
1207  break;
1208 
1209  case LawType::STATIC:
1210  // actions cannot occur in the if or ifcons body
1211  if (
1212  (ifbody && (ifbody->cmask() & sy::ConstantSymbol::Type::M_ACTIONS))
1213  || (ifcons && (ifcons->cmask() & sy::ConstantSymbol::Type::M_ACTIONS))
1214  ) {
1215  config()->error("Actions cannot occur in the \"if\" or \"ifcons\" clauses of static laws.", &head->beginLoc());
1216  return false;
1217  }
